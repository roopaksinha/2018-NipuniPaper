%
% \here
% ~~~~~
% Inserts a line to show where the end of the current work is and 
% where uncompleted work begins.
%
\newcommand{\here}{	
	\color{blue}
	\noindent
	\textbf{<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<=========>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>}
	\color{black}
}   

%
% Insert a comment from Roopak
% ~~~~~~~~~~~~~~~~~~~~~~~~~~~~
% Inserts a comment in blue prefixed by RS -
%
%  \rs{this is a comment from Roopak}

\newcommand{\rs}[1]{          
	\color{blue}
	\noindent
	RS - 
	\textit{{#1}}
	\color{black}
}                              

% Insert a comment from Barry
% ~~~~~~~~~~~~~~~~~~~~~~~~~~~
% Inserts a comment in red prefixed by BRD -
%
%  \brd{this is a comment from Barry}
%
\newcommand{\brd}[1]{         
	\color{red}
	\noindent
	BRD - 
	\textit{{#1}}
	\color{black}
}
